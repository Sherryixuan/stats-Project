library(tidyverse)
library(ggplot2)

youtube_data  <- read_csv("https://docs.google.com/spreadsheets/d/e/2PACX-1vSKjldOzOg5grnUfnfbcdfGuFNKOvbXnC344sZSddtBtwkjWyFbM0X-AOFR2g5MRxTT-S5s7RKOO1jh/pub?output=csv")

glimpse(youtube_data)

# data select
data <- youtube_data %>%
  select(channelName,viewCount,likeCount,duration,datePublished) %>%
  mutate(date = case_when(str_detect(datePublished,"2020") ~ '2020',
                          str_detect(datePublished,"2021") ~ '2021',
                          str_detect(datePublished,"2022") ~ '2022',
                          str_detect(datePublished,"2023") ~ '2023',
                          str_detect(datePublished,"2024") ~ '2024',
                          TRUE ~ "other"),
         channel = ifelse(str_detect(channelName,"Disney"),"Disney","Pixar")
         )
  


# Plot 1: relationship between number of views and the number of likes for each channel (Disney and Pixar)
# get the top 100 views 
sorted_data <- data %>%
  arrange(desc(data$viewCount)) %>%
  slice(1:100) %>%
  select(channel,viewCount,likeCount)

plot1 <- ggplot(sorted_data,aes(x=viewCount,y=likeCount,colour = channel)) + 
  geom_count(pch = 16) +
  scale_color_manual(values = c("Disney" = "skyblue", "Pixar" = "lightpink")) +
  labs(title='top 100 in views and likes') +
  theme(plot.title = element_text(hjust = 0.5)) ## Center the title
ggsave(filename = "plot1.png", plot = plot1)


# Plot 2: relationship between the like count and years, grouped by Disney and Pixar
genre_date <- data %>%
  group_by(date) %>%
  select(channel,date,likeCount) 

plot2 <- genre_date %>%
  ggplot() +
  geom_col(aes(x=likeCount,y=date,fill = channel)) +
  labs(title = "LikeCount V.s. Year",
       y = "year") +
  scale_fill_manual(values = c("Disney" ="#87CEFA", "Pixar" = "pink")) 


ggsave(filename = "plot2.png", plot = plot2)

# Plot3 : mean duration over time for each channel (Disney and Pixar)
summary_data <- data %>%
  group_by(date,channel) %>%
  summarise(mean_duration = mean(duration, na.rm = TRUE))

plot3 <- ggplot(data = summary_data, aes(x = date, y = mean_duration, color = channel, group = channel)) +
  geom_line(linewidth = 1) +  
  geom_point(size = 3) +  
  labs(title = "Mean Duration Over Time by Channel",
       x = "Date",
       y = "Mean Duration") +
  scale_color_manual(values = c("Disney" = "lightgreen", "Pixar" = "thistle"),
                     name = "Channel") 


ggsave(filename = "plot3.png", plot = plot3)

# Plot 4 : the relationship between likeCount and duration based on the group
model <- lm(data$likeCount ~ data$viewCount)
slope <- coef(model)[2]
intercept <- coef(model)[1]

plot4 <- ggplot(data,aes(x=viewCount, y = likeCount)) +
  geom_point(color = "pink") + 
  geom_abline(intercept = intercept, slope = slope, color = "skyblue",size = 3)
ggsave(filename = "plot4.png", plot = plot4)
